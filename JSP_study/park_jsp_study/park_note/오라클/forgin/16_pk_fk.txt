3) 테이블 생성시 외래키 설정
CREATE TABLE USERS(
USERNO NUMBER,
EMAIL VARCHAR2(255) NOT NULL,
PWD VARCHAR2(100) NOT NULL,
NAME VARCHAR2(100) NOT NULL,
PNO VARCHAR2(100) NOT NULL,
ADDRESS VARCHAR2(255)
CONSTRAINT FK_USERS FOREIGN KEY(USERNO)
REFERENCE USERS_INFO(USERNO)
);
